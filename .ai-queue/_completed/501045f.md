# Toast Imperative API (Sonner-Style)

Extend the existing Toast component with a Sonner-style imperative API: `toast()`, `toast.success()`, `toast.error()`, `toast.warning()`, `toast.promise()`.

## Context

### 1. `toast-imperative.tsx`

Create a global store + React bridge pattern:

```typescript
import React, { useCallback, useEffect, useState, useSyncExternalStore } from 'react'

type ToastVariant = 'default' | 'success' | 'error' | 'warning'

interface ToastData {
  id: string
  title: string
  description?: string
  variant: ToastVariant
  duration?: number
  action?: {
    label: string
    altText: string
    onClick: () => void
  }
}

type PromiseToastData<T> = {
  loading: string
  success: string | ((data: T) => string)
  error: string | ((err: unknown) => string)
}
```

**Global store (outside React):**
```typescript
let listeners: Array<() => void> = []
let toasts: ToastData[] = []
let counter = 0

function getSnapshot() { return toasts }
function subscribe(listener: () => void) {
  listeners = [...listeners, listener]
  return () => { listeners = listeners.filter(l => l !== listener) }
}
function emit(toast: ToastData) {
  toasts = [...toasts, toast]
  listeners.forEach(l => l())
}
function dismiss(id: string) {
  toasts = toasts.filter(t => t.id !== id)
  listeners.forEach(l => l())
}
```

**The `toast()` function + variants:**
```typescript
function createToast(title: string, opts?: { description?: string; variant?: ToastVariant; duration?: number; action?: ToastData['action'] }): string {
  const id = `toast-${++counter}`
  emit({ id, title, variant: opts?.variant ?? 'default', description: opts?.description, duration: opts?.duration, action: opts?.action })
  return id
}

// Main function
export function toast(title: string, opts?: { description?: string; duration?: number; action?: ToastData['action'] }): string {
  return createToast(title, { ...opts, variant: 'default' })
}

// Variant shortcuts
toast.success = (title: string, opts?: Omit<Parameters<typeof createToast>[1], 'variant'>): string =>
  createToast(title, { ...opts, variant: 'success' })
toast.error = (title: string, opts?: Omit<Parameters<typeof createToast>[1], 'variant'>): string =>
  createToast(title, { ...opts, variant: 'error' })
toast.warning = (title: string, opts?: Omit<Parameters<typeof createToast>[1], 'variant'>): string =>
  createToast(title, { ...opts, variant: 'warning' })

// Promise helper
toast.promise = <T,>(promise: Promise<T>, data: PromiseToastData<T>): Promise<T> => {
  const id = createToast(data.loading, { variant: 'default' })
  promise
    .then((result) => {
      dismiss(id)
      const msg = typeof data.success === 'function' ? data.success(result) : data.success
      toast.success(msg)
      return result
    })
    .catch((err) => {
      dismiss(id)
      const msg = typeof data.error === 'function' ? data.error(err) : data.error
      toast.error(msg)
    })
  return promise
}

// Dismiss function
toast.dismiss = dismiss
```

**React bridge — `ImperativeToastViewport`:**

A component that subscribes to the global store and renders toasts using the existing Toast compound components. This must be placed inside a `Toast.Provider`.

```typescript
export function ImperativeToastViewport() {
  const currentToasts = useSyncExternalStore(subscribe, getSnapshot, getSnapshot)
  // Render each toast from the store using Toast.Root, Toast.Title, etc.
  // Each toast auto-dismisses after its duration (or provider default)
  // Use the existing Toast.Root variant prop for styling
  return (
    <>
      {currentToasts.map(t => (
        <Toast.Root key={t.id} variant={t.variant} duration={t.duration}>
          <Toast.Title>{t.title}</Toast.Title>
          {t.description && <Toast.Description>{t.description}</Toast.Description>}
          {t.action && (
            <Toast.Action altText={t.action.altText}>
              <button onClick={() => { t.action!.onClick(); dismiss(t.id) }}>
                {t.action.label}
              </button>
            </Toast.Action>
          )}
          <Toast.Close />
        </Toast.Root>
      ))}
    </>
  )
}
```

**IMPORTANT:** Import Toast sub-components from the local `./Toast` file (the existing component), NOT from this file (which would be circular). Use the cast pattern:
```typescript
import { Toast as ToastComponent } from './Toast'
```

Then use `ToastComponent.Root`, `ToastComponent.Title`, etc.

**Also export `useImperativeToasts` hook** that returns the current toast list (for testing or custom rendering):

```typescript
export function useImperativeToasts() {

}
```
