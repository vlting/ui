## Badge Implementation
Replace `styled(Text, ...)` with stl `styled("span", ...)`. Use wrapper function for defaultVariants.

Token mapping:
- `$body` → `$body`, `$1` (fontSize) → `$12`, `$2` → `$14`
- `$color6` → closest stl primary bg token, `$color1` → white/inverse text
- `$color3` → secondary bg, `$color` → default text, `$red10` → `$danger`
- `$borderColor` → `$defaultBorder`, `$color11` → subtle text
- `$10` (borderRadius) → `$full`, `$2` → `$2` (radius)
- paddingHorizontal/paddingVertical → use CSS padding shorthand

```tsx
import { styled } from '../stl-react/src/config'
import type { ComponentProps } from 'react'

const BadgeFrame = styled(
  "span",
  {
    display: "inline-flex",
    alignItems: "center",
    fontFamily: "$body",
    fontSize: "$12",
    fontWeight: "$600",
    borderRadius: "$full",
    overflow: "hidden",
    alignSelf: "flex-start",
  },
  {
    variant: {
      default: { backgroundColor: "$primary", color: "$white" },
      solid: { backgroundColor: "$primary", color: "$white" },
      secondary: { backgroundColor: "$surface2", color: "$defaultText" },
      destructive: { backgroundColor: "$danger", color: "$white" },
      outline: { backgroundColor: "transparent", borderWidth: "1px", borderStyle: "solid", borderColor: "$defaultBorder", color: "$defaultText" },
      subtle: { backgroundColor: "$surface2", color: "$defaultText" },
    },
    size: {
      sm: { fontSize: "$12", paddingLeft: "$1", paddingRight: "$1", paddingTop: "$0", paddingBottom: "$0" },
      md: { fontSize: "$12", paddingLeft: "$2", paddingRight: "$2", paddingTop: "$1", paddingBottom: "$1" },
      lg: { fontSize: "$14", paddingLeft: "$3", paddingRight: "$3", paddingTop: "$1", paddingBottom: "$1" },
    },
  },
  "Badge"
)

export interface BadgeProps extends ComponentProps<typeof BadgeFrame> {
  variant?: 'default' | 'solid' | 'secondary' | 'destructive' | 'outline' | 'subtle'
  size?: 'sm' | 'md' | 'lg'
}

export function Badge({ variant = 'default', size = 'md', ...props }: BadgeProps) {
  return <BadgeFrame variant={variant} size={size} {...props} />
}
```

## Kbd Implementation
Replace `styledHtml('kbd', ...)` with stl `styled("kbd", ...)`. Use wrapper for defaultVariants.

```tsx
import { styled } from '../stl-react/src/config'

const KbdFrame = styled(
  "kbd",
  {
    display: "inline-flex",
    alignItems: "center",
    justifyContent: "center",
    fontFamily: "$mono",
    borderWidth: "1px",
    borderStyle: "solid",
    borderBottomWidth: "2px",
  },
  {
    size: {
      sm: { fontSize: "$12", borderRadius: "$2", minWidth: "20px", paddingLeft: "4px", paddingRight: "4px", paddingTop: "1px", paddingBottom: "1px" },
      md: { fontSize: "$14", borderRadius: "$2", minWidth: "24px", paddingLeft: "6px", paddingRight: "6px", paddingTop: "2px", paddingBottom: "2px" },
    },
  },
  "Kbd"
)

export interface KbdProps {
  children: string
  size?: 'sm' | 'md'
}

export function Kbd({ children, size = 'md' }: KbdProps) {
  return <KbdFrame size={size}>{children}</KbdFrame>
}
```
