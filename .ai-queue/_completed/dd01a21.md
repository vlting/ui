## Core hooks to implement

### `packages/stl-headless/src/useDisclosure.ts`
Generic open/close state with controlled/uncontrolled support.
```ts
interface UseDisclosureProps {
  defaultOpen?: boolean
  open?: boolean           // controlled
  onOpenChange?: (open: boolean) => void
}
interface UseDisclosureReturn {
  isOpen: boolean
  onOpen: () => void
  onClose: () => void
  onToggle: () => void
  getToggleProps: () => { onClick: () => void; 'aria-expanded': boolean }
  getContentProps: () => { hidden: boolean; id: string }
}
```
- Use `useId()` for content ID
- Support controlled mode (when `open` is provided)
- Uncontrolled mode uses internal `useState`

### `packages/stl-headless/src/useListState.ts`
Highlight/active item management for lists with keyboard navigation.
```ts
interface UseListStateProps<T> {
  items: T[]
  defaultHighlightIndex?: number
  onSelect?: (item: T, index: number) => void
  loop?: boolean
}
interface UseListStateReturn<T> {
  highlightIndex: number
  highlightedItem: T | undefined
  setHighlightIndex: (index: number) => void
  getItemProps: (index: number) => { 'aria-selected': boolean; onMouseEnter: () => void; onClick: () => void }
  onKeyDown: (e: React.KeyboardEvent) => void  // Arrow, Home, End, Enter
}
```
- Arrow Up/Down changes highlight
- Home/End jumps to first/last
- Enter selects
- Optional loop (wrap around)

### `packages/stl-headless/src/useSearch.ts`
Search/filter state with highlight index reset.
```ts
interface UseSearchProps<T> {
  items: T[]
  filterFn: (item: T, query: string) => boolean
}
interface UseSearchReturn<T> {
  query: string
  setQuery: (q: string) => void
  filtered: T[]
  inputProps: { value: string; onChange: (e: ChangeEvent<HTMLInputElement>) => void }
}
```
